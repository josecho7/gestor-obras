// ─── sync con Google Sheets ───────────────────────────────
async function loadData(){
  state.syncStatus="loading";render();
  try{
    const res=await fetch(SCRIPT_URL,{method:"GET",redirect:"follow"});
    const json=await res.json();
    if(json.ok){
      state.tareas=json.data.tareas||[];
      state.rubros=json.data.rubros||DEFAULT_RUBROS;
      state.obras=json.data.obras||[];
      state.syncStatus="ok";
    } else { state.syncStatus="error"; }
  } catch(e){ state.syncStatus="error"; }
  render();
}

async function saveData(){
  state.syncStatus="saving";render();
  try{
    const body=JSON.stringify({tareas:state.tareas,rubros:state.rubros,obras:state.obras});
    const res=await fetch(SCRIPT_URL,{method:"POST",body,redirect:"follow"});
    const json=await res.json();
    state.syncStatus=json.ok?"ok":"error";
  } catch(e){ state.syncStatus="error"; }
  render();
}

async function submitForm(){
  if(!state.form.titulo.trim()){alert("El título es obligatorio");return}
  if(state.editId){
    state.tareas=state.tareas.map(t=>t.id===state.editId?{...t,...state.form}:t);
  } else {
    state.tareas=[...state.tareas,{...state.form,id:Date.now(),fechaCreacion:new Date().toISOString(),tiemposEjecucion:[]}];
  }
  state.showForm=false;state.editId=null;state.form=emptyForm();
  await saveData();toast("Tarea guardada")
}

async function cambiarEstado(id,estado){
  if(estado==="Completada"){state.tiempoModal=id;state.tiempoInput="";render();return}
  state.tareas=state.tareas.map(t=>t.id===id?{...t,estado}:t);
  await saveData()
}

async function confirmarTiempo(){
  const hrs=parseFloat(state.tiempoInput);
  state.tareas=state.tareas.map(t=>t.id===state.tiempoModal?{
    ...t,estado:"Completada",fechaCompletada:new Date().toISOString(),
    tiemposEjecucion:[...(t.tiemposEjecucion||[]),{horas:isNaN(hrs)?null:hrs,fecha:new Date().toISOString()}]
  }:t);
  state.tiempoModal=null;
  await saveData();toast("Tarea completada")
}

async function eliminarTarea(id){

function editarTarea(id){

  render();window.scrollTo(0,0)
}

async function agregarRubro(){
  const r=state.nuevoRubro.trim();
  if(!r)return;
  if(state.rubros.includes(r)){alert("Ese rubro ya existe");return}
  state.rubros=[...state.rubros,r];state.nuevoRubro="";state.showRubroInput=false;
  await saveData();toast("Rubro agregado")
}

async function eliminarRubro(r){
  if(state.tareas.some(t=>t.rubro===r)){alert("No se puede eliminar un rubro con tareas asignadas.");return}
  if(!confirm(`¿Eliminar rubro "${r}"?`))return;
  state.rubros=state.rubros.filter(x=>x!==r);
  await saveData();toast("Rubro eliminado")
}

async function agregarObra(){
  const nombre=state.nuevaObra.trim();
  if(!nombre)return;
  state.obras=[...state.obras,{id:Date.now().toString(),nombre}];
  state.nuevaObra="";state.showObraInput=false;
  await saveData();toast("Obra agregada")
}

async function eliminarObra(id){

function importarJSON(){
  const inp=document.createElement("input");inp.type="file";inp.accept=".json";
  inp.onchange=async e=>{
    const file=e.target.files[0];if(!file)return;
    const reader=new FileReader();
    reader.onload=async ev=>{
      try{
        const data=JSON.parse(ev.target.result);
        if(!data.tareas||!data.rubros||!data.obras){alert("Archivo inválido");return}
        if(!confirm(`¿Importar ${data.tareas.length} tareas, ${data.obras.length} obras y ${data.rubros.length} rubros?\nEsto REEMPLAZA los datos actuales.`))return;
        state.tareas=data.tareas;state.rubros=data.rubros;state.obras=data.obras;
        await saveData();toast("Datos importados")
      }catch{alert("Error al leer el archivo")}
    };reader.readAsText(file)
  };inp.click()
}

// ─── stats helpers ────────────────────────────────────────
function statsGroup(tareas,keyFn,labelFn){
  const g={};
  tareas.forEach(t=>{
    (t.tiemposEjecucion||[]).filter(te=>te.horas!=null).forEach(te=>{
      const k=keyFn(t);if(!k)return;
      if(!g[k])g[k]={label:labelFn(t),horas:[],count:0};
      g[k].horas.push(te.horas);g[k].count++;
    })
  });
  return Object.values(g).map(g=>{
    const sum=g.horas.reduce((a,b)=>a+b,0);
    return{...g,total:sum,prom:sum/g.horas.length,min:Math.min(...g.horas),max:Math.max(...g.horas)}
  }).sort((a,b)=>b.total-a.total)
}

function renderAnalisis(){
  const completadas=state.tareas.filter(t=>t.estado==="Completada");
  const selObra=`<select style="width:auto" onchange="state.cmpObra=this.value;render()"><option value="">Todas las obras</option>${state.obras.map(o=>`<option value="${esc(o.id)}"${state.cmpObra===o.id?" selected":""}>${esc(o.nombre)}</option>`).join("")}</select>`;
  const selRubro=`<select style="width:auto" onchange="state.cmpRubro=this.value;render()"><option value="">Todos los rubros</option>${state.rubros.map(r=>`<option value="${esc(r)}"${state.cmpRubro===r?" selected":""}>${esc(r)}</option>`).join("")}</select>`;
  const selPers=`<select style="width:auto" onchange="state.cmpPers=this.value;render()"><option value="">Todas las personas</option>${responsables().map(r=>`<option value="${esc(r)}"${state.cmpPers===r?" selected":""}>${esc(r)}</option>`).join("")}</select>`;
  let base=completadas;
  if(state.cmpObra)  base=base.filter(t=>t.obraId===state.cmpObra);
  if(state.cmpRubro) base=base.filter(t=>t.rubro===state.cmpRubro);
  if(state.cmpPers)  base=base.filter(t=>t.responsable===state.cmpPers);

  function tbl(rows,title){
    if(!rows.length)return`<p style="color:#888;font-size:13px">Sin datos suficientes.</p>`;
    const mx=Math.max(...rows.map(r=>r.total));
    return`<table><thead><tr><th>${title}</th><th>Ejecuciones</th><th>Total (h)</th><th>Promedio (h)</th><th>Mín</th><th>Máx</th><th style="width:120px">Proporción</th></tr></thead><tbody>
      ${rows.map(r=>{const p=mx>0?Math.round((r.total/mx)*100):0;return`<tr><td style="font-weight:500">${esc(r.label)}</td><td>${r.count}</td><td>${r.total.toFixed(1)}</td><td>${r.prom.toFixed(1)}</td><td>${r.min}</td><td>${r.max}</td><td><div style="display:flex;align-items:center;gap:6px"><div class="bar-wrap"><div class="bar" style="width:${p}%"></div></div><span style="font-size:11px;color:#888">${p}%</span></div></td></tr>`}).join("")}
    </tbody></table>`
  }

  const grupos={};
  base.filter(t=>t.tiemposEjecucion?.some(te=>te.horas!=null)).forEach(t=>{
    const key=`${t.obraId||""}||${t.rubro}||${t.titulo.toLowerCase().trim()}`;
    if(!grupos[key])grupos[key]={titulo:t.titulo,rubro:t.rubro,obraId:t.obraId||"",entradas:[]};
    t.tiemposEjecucion.forEach(te=>{if(te.horas!=null)grupos[key].entradas.push({horas:te.horas,fecha:te.fecha})});
  });
  const gruposArr=Object.values(grupos);

  function sec(title,content){return`<div style="margin-bottom:24px"><div style="font-size:15px;font-weight:500;margin-bottom:10px;padding-bottom:6px;border-bottom:1px solid #eee">${title}</div>${content}</div>`}

  return`<div>
    <div class="row" style="margin-bottom:18px">${selObra}${selRubro}${selPers}</div>
    ${sec("Tiempo por rubro",tbl(statsGroup(base,t=>t.rubro,t=>t.rubro),"Rubro"))}
    ${sec("Tiempo por tipo de tarea",tbl(statsGroup(base,t=>t.tipo,t=>t.tipo),"Tipo"))}
    ${sec("Tiempo por persona",tbl(statsGroup(base,t=>t.responsable,t=>t.responsable),"Persona"))}
    ${sec("Comparativo por tarea similar",gruposArr.length===0
      ?`<div class="card" style="text-align:center;color:#888;padding:1.5rem">Sin datos aún.</div>`
      :`<div style="display:flex;flex-direction:column;gap:12px">${gruposArr.map(g=>{
          const hrs=g.entradas.map(e=>e.horas);
          const prom=hrs.reduce((a,b)=>a+b,0)/hrs.length;
          const mn=Math.min(...hrs),mx=Math.max(...hrs);
          return`<div class="card">
            <div style="display:flex;justify-content:space-between;flex-wrap:wrap;gap:10px;margin-bottom:10px">
              <div><div style="font-weight:500;font-size:14px;margin-bottom:4px">${esc(g.titulo)}</div>
              <div class="row" style="gap:5px">${g.rubro?badge(g.rubro,"#F1EFE8","#5F5E5A"):""}${g.obraId?badge(obraName(g.obraId),"#EEEDFE","#3C3489"):""}</div></div>
              <div style="display:flex;gap:8px;flex-wrap:wrap">
                ${[["Promedio",prom.toFixed(1)+"h","#EAF3DE","#3B6D11"],["Mínimo",mn+"h","#E6F1FB","#185FA5"],["Máximo",mx+"h","#FAEEDA","#BA7517"]].map(([l,v,bg,tc])=>`<div style="text-align:center;background:${bg};border-radius:8px;padding:6px 12px"><div style="font-size:11px;color:${tc}">${l}</div><div style="font-size:16px;font-weight:500;color:${tc}">${v}</div></div>`).join("")}
              </div>
            </div>
            <div class="row" style="gap:6px;flex-wrap:wrap">
              ${g.entradas.map((e,j)=>`<span style="font-size:12px;padding:3px 10px;border-radius:6px;background:#f5f5f5;border:1px solid #eee">#${j+1}: ${e.horas}h <span style="color:#bbb">(${new Date(e.fecha).toLocaleDateString("es-PY")})</span></span>`).join("")}
            </div>
            ${hrs.length>1?`<div style="margin-top:10px"><div style="display:flex;gap:3px;height:10px;border-radius:4px;overflow:hidden">${hrs.map(h=>{const p=mx>mn?((h-mn)/(mx-mn))*100:50;return`<div style="flex:1;background:hsl(${120-p*1.2},55%,45%)" title="${h}h"></div>`}).join("")}</div><div style="display:flex;justify-content:space-between;font-size:11px;color:#bbb;margin-top:3px"><span>más rápido</span><span>más lento</span></div></div>`:""}
          </div>`
        }).join("")}</div>`
    )}
  </div>`
}
